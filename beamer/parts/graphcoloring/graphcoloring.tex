\begin{frame}
    \frametitle{The Four Color Theorem}
    \begin{theorem}
        Every planar graph can be colored in at most four colors.
    \end{theorem}

    A simple statement. First formulated in 1852, proven over a hundred years later in 1976. 
\end{frame}
